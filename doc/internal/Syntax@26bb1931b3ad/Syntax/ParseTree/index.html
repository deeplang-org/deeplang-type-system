Module Syntax.ParseTree

This module defines the abstract syntax of deeplang. The syntax tree will be used for all subsequent compiler passes, until code generation.

  1. | Symbol of int

a globally unique id for variables

module NodeId : sig ... end

NodeId contains types for unique ids for various kinds of AST nodes. Can be used to attach extra data to AST nodes, via an external table.

type typ = {
  1. shape : typ_shape;
  2. span : SyntaxError.src_span;
}

For types internally generated by the compiler, span should be None.

and typ_shape =
  1. | TyVar of string
  2. | TyUnit
  3. | TyBool
  4. | TyInt of int_typ_sign * int_typ_size
  5. | TyFloat of float_typ_size
  6. | TyChar
  7. | TyThis
  8. | TyArray of typ * int
  9. | TyTuple of typ list
  10. | TyNamed of typ_name * typ list

typ_shape declares the structure of deeplang types

Expressions

type unary_op =
  1. | UnOpNeg
  2. | UnOpNot
type compare_op =
  1. | BinOpLt
  2. | BinOpLeq
  3. | BinOpGt
  4. | BinOpGeq
  5. | BinOpEq
  6. | BinOpNeq
    (*

    note : BinOpLNot --modify--> BinOpLXor * optional : | BinOpBOr | BinOpBAnd | BinOpBXor * B = bit, | '|' | '&' | '^' * it might be useful for IoT development

    *)
type calculate_op =
  1. | BinOpLOr
  2. | BinOpLAnd
  3. | BinOpLXor
  4. | BinOpBOr
  5. | BinOpBAnd
  6. | BinOpBXor
  7. | BinOpLShift
  8. | BinOpRShift
  9. | BinOpAdd
  10. | BinOpSub
  11. | BinOpMul
  12. | BinOpDiv
  13. | BinOpMod
    (*

    Some notes on BinOpAssign:

    • BinOpAssign None is normal assignment
    • BinOpAssign Add is "+=", BinOpAssign Sub is "-=", etc.
    • Not all operators have corresponding assignment operators in the parser. This is just a simplification of AST definition
    *)
type binary_op =
  1. | BinOpCompare of compare_op
  2. | BinOpCalculate of calculate_op
type expr = {
  1. shape : expr_shape;
  2. expr_id : NodeId.expr;
  3. span : SyntaxError.src_span;
}

  1. | ExpLit of literal
  2. | ExpVar of variable
  3. | ExpUnOp of unary_op * expr
  4. | ExpBinOp of binary_op * expr * expr
  5. | ExpTuple of expr list
  6. | ExpADT of adt_label * expr list
  7. | ExpStruct of typ_name * (struct_field * expr) list

  10. | ExpApp of func_name * expr list
  11. | ExpMethod of expr * method_name * expr list
  12. | ExpIf of expr * expr * expr
  13. | ExpMatch of expr * (pattern * expr) list

  1. gvar_name : variable;

  4. gvar_value : expr;

type func_impl = func_decl * stmt
type struct_def_field_attr =
  1. | Struct_Field
  2. | Struct_Delegate
type struct_def_field = struct_field * typ * struct_def_field_attr
type struct_def = {
  1. struct_name : typ_name;
  2. struct_fields : struct_def_field list;
}
type adt_def = {
  1. adt_name : typ_name;
  2. adt_branches : (adt_label * typ list) list;
}
type interface_decl = {
  1. intf_decl_name : intf_name;
  2. intf_decl_methods : func_decl list;
}
type methods_impl = {
  1. impl_intf : intf_name option;
  2. impl_typ : typ_name;
  3. impl_id : NodeId.impl;
  4. impl_methods : func_impl list;
}
type top_clause = {
  1. shape : top_clause_shape;
  2. span : SyntaxError.src_span;
}
and top_clause_shape =
  1. | StructDef of struct_def
  2. | ADTDef of adt_def
  3. | InterfaceDecl of interface_decl
  4. | MethodsImpl of methods_impl
  5. | FunctionDef of func_impl
  6. | GlobalVarDef of global_variable_def
